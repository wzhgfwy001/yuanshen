### 代码调用示例
```javascript
const registry = require('./agency-registry/registry.json');

// 查找匹配的Agent
function findAgent(task) {
    const t = task.toLowerCase();
    return Object.values(registry.agents).find(a => 
        a.trigger_keywords.some(kw => t.includes(kw.toLowerCase()))
    );
}
```
