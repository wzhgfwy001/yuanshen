### 3.3 性能测试脚本

```python
# test_performance.py
import time
from services import DataService, FilterService
from models import Student, FilterConfig

def test_import_performance():
    """测试导入性能"""
    ds = DataService()
    
    # 测试学校导入
    start = time.time()
    ds.import_schools_excel("test_data/large_schools.xlsx")
    elapsed = time.time() - start
    print(f"导入 {len(ds.schools)} 条学校数据：{elapsed:.2f}秒")
    
    # 测试学生导入
    start = time.time()
    ds.import_students_excel("test_data/large_students.xlsx")
    elapsed = time.time() - start
    print(f"导入 {len(ds.students)} 个学生：{elapsed:.2f}秒")

def test_filter_performance():
    """测试筛选性能"""
    ds = DataService()
    ds.import_schools_excel("test_data/large_schools.xlsx")
    ds.import_students_excel("test_data/students.xlsx")
    
    fs = FilterService(ds)
    student = ds.students[0]
    config = FilterConfig()
    
    start = time.time()
    results = fs.filter(student, config)
    elapsed = time.time() - start
    
    print(f"筛选 {len(results)} 条结果：{elapsed:.2f}秒")

def test_batch_filter_performance():
    """测试批量筛选性能"""
    ds = DataService()
    ds.import_schools_excel("test_data/large_schools.xlsx")
    ds.import_students_excel("test_data/large_students.xlsx")
    
    fs = FilterService(ds)
    config = FilterConfig()
    
    start = time.time()
    results_dict = fs.batch_filter(ds.students, config)
    elapsed = time.time() - start
    
    total_results = sum(len(r) for r in results_dict.values())
    print(f"批量筛选 {len(ds.students)} 个学生，共 {total_results} 条结果：{elapsed:.2f}秒")

if __name__ == "__main__":
    test_import_performance()
    test_filter_performance()
    test_batch_filter_performance()
```
